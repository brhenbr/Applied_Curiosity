DROP TABLES TB_RELATORIO;


-- ITEM 1 - CRIACAO DAS TABELAS

CREATE TABLE TB_TP_DOMINIO (
       ID_TP_DOMINIO   INTEGER    NOT NULL,
       DESC_TP_DOMINIO CHAR(30)   NOT NULL,
       DEF_COBOL       CHAR(2)    NOT NULL,
       PRIMARY KEY (ID_TP_DOMINIO));

CREATE TABLE TB_DOMINIO (
       ID_DOMINIO      INTEGER    NOT NULL,
       NOM_DOMINIO     CHAR (50)  NOT NULL,
       ID_TP_DOMINIO   INTEGER    NOT NULL,
       POS_INI_DOMINIO INTEGER    NOT NULL,
       POS_FIM_DOMINIO INTEGER    NOT NULL,
       PRIMARY KEY (ID_DOMINIO),
       FOREIGN KEY (ID_TP_DOMINIO) REFERENCES TB_TP_DOMINIO (ID_TP_DOMINIO) ON DELETE RESTRICT);

CREATE TABLE TB_DT_MESTRE (
       ID_DADO      INTEGER IDENTITY GENERATED ALWAYS NOT NULL,
       DT_MOV       DATE                              NOT NULL,
       CD_PROD      INTEGER                           NOT NULL,
       VL_MOV       DECIMAL(15,2)                     NOT NULL,
       TP_MOV       INTEGER                           NOT NULL,
       PRIMARY KEY (ID_DADO),
       FOREIGN KEY (TP_MOV) REFERENCES TB_TP_MOV (ID_TP_MOV) ON DELETE RESTRICT);

CREATE TABLE TB_TP_MOV (
       ID_TP_MOV     INTEGER  NOT NULL,
       DESC_TP_MOV   CHAR(15) NOT NULL,
       DEF_NUMERICA  CHAR(1)  NOT NULL,
       PRIMARY KEY (ID_TP_MOV));

CREATE TABLE TB_RELAT-DOMINIO (
       ID_RELATORIO      INTEGER                              NOT NULL,
       ID_DOMINIO   INTEGER                           NOT NULL,
       IND_SOMA CHAR(1)                               NOT NULL,
       IND_ORDENA   INTEGER                           NOT NULL,
       PRIMARY KEY (ID_RELATORIO,ID_DOMINIO),
       FOREIGN KEY (ID_RELATORIO) REFERENCES TB_RELATORIO (ID_RELATORIO) ON DELETE RESTRICT,
       FOREIGN KEY (ID_DOMINIO)   REFERENCES TB_DOMINIO   (ID_DOMINIO)   ON DELETE RESTRICT);
